import requests  # 导入requests
from bs4 import BeautifulSoup  # 导入bs4中的BeautifulSoup
import json # json.loads(string)可以将字符串专为字典
import pandas # 整理数据，是数据展示的直观有条理

# 第一步：模拟登陆

s = requests.Session()
'''Session会话对象可以跨请求保持某些参数，即在你使用该对象登陆某个网站后，
   如果你还访问该网站的其他网页都会默认使用该对象之前使用的cookie等参数
'''
username = input("请输入学号：")
password = input("请输入密码：")

login_url = 'https://cas.gzhu.edu.cn/cas_server/login' # 登陆界面中登陆请求的地址
headers = {
     'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/71.0.3578.98 Safari/537.36'
}
html = s.get(login_url, headers = headers)
soup = BeautifulSoup(html.content, 'lxml')
lt = soup.select('input[name="lt"]')[0].get('value')
execution = soup.select('input[name="execution"]')[0].get('value')

data = {
  'username': username,
  'password': password,
   'captcha':'' , 
  'warn': 'true',
  'lt': lt,
  'execution': execution,
  '_eventId': 'submit',
  'submit': '登录'
}

r = s.post(login_url, data=data,headers=headers,allow_redirects=False)
resp = s.get('http://my.gzhu.edu.cn/', headers=headers)
resp.encoding = 'utf-8'

# 第二步：进入新教育系统

url1 = 'http://jwxt.gzhu.edu.cn/jwglxt/lyiotlogin'
result = s.get(url1)

score_url = "http://jwxt.gzhu.edu.cn/jwglxt/cjcx/cjcx_cxDgXscj.html?doType=query&gnmkdm"
sform_data = {
    'xnm':'2018',
    'xqm':'3',
    '_search': 'false',
    'queryModel.showCount':'15',
    'queryModel.currentPage':'1',
    'queryModel.sortName': '',
    'queryModel.sortOrder':'asc'

}
score = s.post(score_url,headers=headers)
soup = BeautifulSoup(score.content,'lxml')
# print(soup.text)
# score = s.post(score_url,sform_data)
# score = BeautifulSoup(score.content,'lxml')

score = json.loads(score.text)
score = score["items"]
cjList = [] # 成绩列表
for item in score:
    temp = {}
    temp["xuenian"] = item["xnmmc"]
    temp["xueqi"] = item["xqmmc"]
    temp["courseName"] = item["kcmc"]
    temp["score"] = item["cj"]
    temp["jidian"] = item["jd"]
    temp["teacher"] = item["jsxm"]
    cjList.append(temp)
sortS = pandas.DataFrame(cjList)

# table_url 是学生课表查询界面中点击“查询”后发送请求的地址 
table_url = 'http://jwxt.gzhu.edu.cn/jwglxt/kbcx/xskbcx_cxXsKb.html?'
tform_data = {
    'xnm' :'2018',
    'xqm' :'12'
}
result = s.post(table_url,tform_data)
result = BeautifulSoup(result.content,'lxml')
result = json.loads(result.text)
result = result["kbList"]
kbList = [] # 课表列表
for item in result:
    temp = {}
    temp["courseName"] = item["kcmc"]
    temp["time"] = "(" + item["jc"] + ")" + item["zcd"]
    temp["address"] = item["xqmc"] + " " + item["cdmc"]
    temp["teacherName"] = item["xm"]
    temp["week"] = item["xqjmc"]
    temp["team"] = item["jxbmc"]
    kbList.append(temp)
df = pandas.DataFrame(kbList) # 通过pandas对数据进行整理，从而是其展示更有条理清晰
df       # 展示出课表
sortS    # 展示出成绩

注意点

  • 建立一个Session 会话对象，并且在之后的发送请求时都是通过该对象进行。Session 可以记录用户在当前网站的登陆状态。每次向服务器发送请求时，服务器都会检查cookies里相应的会话，从而判断用户的状态，如果会话到期或者无效，则会跳转到登陆界面、其他界面或者收到错误的响应信息。

  • 在模拟登陆网站时，要注意表单中需要提交的值，有些值时变化的，所以每次登陆时，我们需要先获取到这些变值，然后再一次发送登陆请求。以在广州大学统一身份认证平台登陆为例。

如上图，lt , execution 都是不固定的。通过发送get请求获取登录界面中lt, execution的值同时也会记录此时的cookies值。
